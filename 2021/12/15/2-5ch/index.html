cubes = [i**3 for i in range(1,12)]
print(cubes)

print('The first three items in the list are:')
print(cubes[:3])

print('The last three items in the list are:')
last_three_index = len(cubes) - 3
print(cubes[last_three_index:])

#打印“中间”三个元素，判断一下“中间”
#为了配合“中间”，将列表调整为11个元素
def middle():
    l = len(cubes)
    if l % 2 == 1:
        return int((l + 1) / 2)
    elif l % 2 == 0:
        return int(l / 2)
mid = middle() - 1  #计数从0开始

print('Three items from the middle of the list are:')
print(cubes[mid - 1], cubes[mid], cubes[mid + 1])
